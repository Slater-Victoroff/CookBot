CN.hearstQue = (function () {
  var queue = [],
    alreadyDone = false,
    order = 0;
  return {
    register: function (fn) {
      if (!fn) {
        return;
      }
      queue.push(fn);
      if (alreadyDone) {
        this.run(window.pageAds);
      }
    },
    getQue: function () {
      return queue;
    },
    run: function (pageAds) {
      while (queue.length > 0) {
        queue[0](pageAds);
        queue.shift();
      }
      alreadyDone = true;
    }
  };
})();

                        <b style="display: block;margin-bottom: 5px;">NUTRIENT BALANCE INDICATOR&trade;</b>
